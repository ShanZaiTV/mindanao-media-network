document.addEventListener("DOMContentLoaded", () => {
  console.log("Mindanao Media Network CATV Loaded");

  // Sample channel list
  const channels = [
    {
      name: "MMN News",
      category: "News",
      status: "LIVE"
    },
    {
      name: "MMN Sports",
      category: "Sports",
      status: "LIVE"
    },
    {
      name: "MMN Movies",
      category: "Entertainment",
      status: "OFFLINE"
    },
    {
      name: "MMN Music",
      category: "Music",
      status: "LIVE"
    }
  ];

  // Container
  const channelContainer = document.getElementById("channel-list");

  // Display channels
  function renderChannels() {
    channelContainer.innerHTML = "";

    channels.forEach((channel, index) => {
      const card = document.createElement("div");
      card.classList.add("channel-card");

      card.innerHTML = `
        <div class="channel-header">
          <h3>${channel.name}</h3>
          <span class="${channel.status.toLowerCase()}">
            ${channel.status}
          </span>
        </div>

        <p>Category: ${channel.category}</p>

        <button class="watch-btn" data-index="${index}">
          Watch Channel
        </button>
      `;

      channelContainer.appendChild(card);
    });
  }

  // Watch button event
  channelContainer.addEventListener("click", (event) => {
    if (event.target.classList.contains("watch-btn")) {
      const index = event.target.dataset.index;
      const selectedChannel = channels[index];

      alert(
        `Now Playing: ${selectedChannel.name}\nCategory: ${selectedChannel.category}`
      );

      console.log("Streaming:", selectedChannel.name);
    }
  });

  // Digital Clock
  function updateClock() {
    const clock = document.getElementById("live-clock");

    if (clock) {
      const now = new Date();

      const time = now.toLocaleTimeString();
      const date = now.toLocaleDateString();

      clock.innerHTML = `
        <strong>${time}</strong><br>
        <small>${date}</small>
      `;
    }
  }

  setInterval(updateClock, 1000);
  updateClock();

  // Simulated Breaking News Ticker
  const tickerMessages = [
    "Welcome to Mindanao Media Network CATV",
    "24/7 Local & International Channels",
    "Breaking News Updates Every Hour",
    "Enjoy HD Entertainment and Sports"
  ];

  let tickerIndex = 0;

  function updateTicker() {
    const ticker = document.getElementById("news-ticker");

    if (ticker) {
      ticker.textContent = tickerMessages[tickerIndex];

      tickerIndex++;

      if (tickerIndex >= tickerMessages.length) {
        tickerIndex = 0;
      }
    }
  }

  setInterval(updateTicker, 4000);
  updateTicker();

  // Initialize channel rendering
  renderChannels();
